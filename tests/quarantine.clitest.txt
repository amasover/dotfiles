--- metapac-declared: hostname_groups -> group files -> sorted union ---

$ t=$(mktemp -d) && mkdir -p "$t/groups" && printf '[hostname_groups]\n"testhost" = [\n  "alpha",\n  "%s/local.toml",\n]\n' "$t" > "$t/config.toml" && printf '[arch]\npackages = [\n  "zeta-pkg",\n  "alpha-pkg",\n]\n' > "$t/groups/alpha.toml" && printf '[arch]\npackages = [\n  "local-pkg",\n  "alpha-pkg",\n]\n' > "$t/local.toml" && .local/bin/tools/metapac-declared --config-dir "$t" --hostname testhost
alpha-pkg
local-pkg
zeta-pkg
$
